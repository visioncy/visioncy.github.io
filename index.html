yum -y install build-essential asciidoc binutils bzip2 gawk gettext git libncurses5-dev libz-dev patch unzip zlib1g-dev lib32gcc1 libc6-dev-i386 subversion flex uglifyjs git-core gcc-multilib p7zip p7zip-full msmtp libssl-dev texinfo libglib2.0-dev xmlto qemu-utils upx libelf-dev autoconf automake libtool autopoint device-tree-compiler

#添加USB相关支持

Kernel modules —> USB Support —> <*> kmod-usb-ohci.

3.开始编译

#先同步软件包(启动20个线程，加快下载，这样就不用等10个小时了，我用了50分钟下载完毕)：
make download  V=s -j20  
#开始编译(第一次编译，不启用多线程。二次编译时可以选-j4,这个考虑你的CPU支持的线程数)
make -j1 V=s

错误：

configure: error: you should not run configure as root (set FORCE_UNSAFE_CONFIGURE=1 in environment to bypass this check)

解决方法A：

sudo make FORCE_UNSAFE_CONFIGURE=1 -j1 V=s

解决方法B：

export FORCE_UNSAFE_CONFIGURE=1
source /etc/profile
然后：
su –root  # 切换到root账号
make V=99 # 编译

错误：

tar: This does not look like a tar archive

解决方法：（删除不能解压的包重新运行make）

4.刷机

#编译后的目录
/home/lede/bin/targets/ramips/mt7621

openwrt 配置ip跳转

发表于 2020-03-12

1. 命令配置（重启后失效）

#设置DNAT 使用指定访问ip跳转
iptables -t nat -A PREROUTING -p tcp -m tcp -d 31.105.78.1 --dport 8087 -j DNAT --to-destination 112.7.29.1:8080
#打印路由设定
iptables -t nat -L --line-number
#删除第3行PREROUTING
iptables -t nat -D PREROUTING 3
#恢复默认
iptables -t nat -F PREROUTING

2. 开机自动加载（持久化设置）

#编辑文件
vi /etc/firewall.user
#加入配置
iptables -t nat -A PREROUTING -p tcp -m tcp -d 31.105.78.1 --dport 8087 -j DNAT --to-destination 112.7.29.1:8080
#重启网络
/etc/init.d/network restart

[mongodb-org-4.2]
name=MongoDB Repository
baseurl=https://repo.mongodb.org/yum/redhat/$releasever/mongodb-org/4.2/x86_64/
gpgcheck=1
enabled=1
gpgkey=https://www.mongodb.org/static/pgp/server-4.2.asc

2.安装mongodb常用包

sudo yum install -y mongodb-org

移动数据库

#复制数据库
rsync -av /var/lib/mongo /mnt/database/mongodb/
#备份旧数据库
sudo mv /var/lib/mongo /var/lib/mongo.bak

如果运行数据库是提示没权限，可以进行授权

#Change owner and group
sudo chown -R mongod:mongod /home/mongodb/mongo
#Change context
sudo chcon -R -u system_u -t mongod_var_lib_t /home/mongodb/mongo/

3.修改数据库保存路径
/etc/mongod.conf

初步化数据库
mongo –dbpath /path/

添加超级管理员

mongo

  {
    user: "xhadmin",
    pwd: "abc123",
    roles: [ { role: "userAdminAnyDatabase", db: "admin" }, "readWriteAnyDatabase" ]
  }
)

openfire 优化

发表于 2020-02-19

原来连接与登录共收发20条信息
修改后连接与登录共收发8条信息

1.[Server Settings]->[Registration & Login]->[SASL Mechanisms]只保留[SCRAM-SHA-1]（暂时测试不确认是否有优化）
并且xmpp.client.tls.policy改为:disabled

  • required 必需
  • optional 可选
  • disabled 关闭
  • legacyMode 一种要求连接立即加密的策略（与“必需”策略相反，它允许最初未加密的连接通过StartTLS加密。

这样可以减少一次接收信息

16:19:20 RECV (0): <stream:features><starttls xmlns="urn:ietf:params:xml:ns:xmpp-tls"></starttls><mechanisms xmlns="urn:ietf:params:xml:ns:xmpp-sasl"><mechanism>PLAIN</mechanism><mechanism>SCRAM-SHA-1</mechanism><mechanism>CRAM-MD5</mechanism><mechanism>DIGEST-MD5</mechanism></mechanisms><compression xmlns="http://jabber.org/features/compress"><method>zlib</method></compression><ver xmlns="urn:xmpp:features:rosterver"/><register xmlns="http://jabber.org/features/iq-register"/><c xmlns="http://jabber.org/protocol/caps" hash="sha-1" node="https://www.igniterealtime.org/projects/openfire/" ver="RoQdsUcljslH4bMYE/tqetK2DQU="/></stream:features>

2.文本已经加密，禁用TLS

3.SaslMechanism使用”PLAIN”

centOS 安装 nginx

发表于 2019-11-08

1. 下载nginx包

sudo wget -c https://nginx.org/download/nginx-1.17.5.tar.gz
sudo tar -zxvf nginx-1.17.5.tar.gz
cd nginx-1.17.5

2. 配置与编译

sudo yum -y install pcre-devel openssl openssl-devel
sudo yum -y install perl-devel perl-ExtUtils-Embed

sudo ./configure --with-stream --with-http_ssl_module --with-http_perl_module
sudo make
sudo make install

3. 设置开机启动

sudo vim /etc/systemd/system/nginx.service

文件内容

[Unit]
Description=nginx service
After=network.target 

[Service] 
Type=forking 
ExecStart=/usr/local/nginx/sbin/nginx
ExecReload=/usr/local/nginx/sbin/nginx -s reload
ExecStop=/usr/local/nginx/sbin/nginx -s quit
PrivateTmp=true 

[Install] 
WantedBy=multi-user.target

设置自动启动

sudo systemctl enable nginx.service

4.手动启动测试代码

cd /usr/local/nginx/sbin/
./nginx 
./nginx -s stop
./nginx -s quit
./nginx -s reload

centOS 安装 turnserver

发表于 2019-11-08

1. 下载turnserver包

wget http://turnserver.open-sys.org/downloads/v4.5.1.1/turnserver-4.5.1.1.tar.gz 
sudo tar -zxvf turnserver-4.5.1.1.tar.gz
cd turnserver-4.5.1.1 && sudo ./configure

2. 配置与编译

sudo yum install -y make gcc cc gcc-c++ wget
sudo yum install -y openssl-devel libevent libevent-devel mysql-devel mysql-server
sudo make && sudo make install

3. 修改配置文件

sudo cp /usr/local/etc/turnserver.conf.default   /usr/local/etc/turnserver.conf
sudo vim /usr/local/etc/turnserver.conf

在文件顶部加入如下配置 

listening-port=3478
user=mangoturn:cvKuUdaQ
realm=turn02.imangoim.com

附：手去启动TURN命令

turnserver -c /usr/local/etc/turnserver.conf -o –v

4.设置开机启动

sudo vim /etc/systemd/system/turnserver.service

文件内容

[Unit]
Description=coturn
Documentation=man:coturn(1) man:turnadmin(1) man:turnserver(1)
After=syslog.target network.target

[Service]
#User=turnserver
#Group=turnserver
Type=forking
#EnvironmentFile=/etc/sysconfig/turnserver
PIDFile=/var/run/turnserver.pid
ExecStart=/usr/local/bin/turnserver -o -c /usr/local/etc/turnserver.conf $EXTRA_OPTIONS
ExecStopPost=/usr/bin/rm -f /var/run/turnserver.pid
Restart=on-abort

LimitCORE=infinity
LimitNOFILE=999999
LimitNPROC=60000
LimitRTPRIO=infinity
LimitRTTIME=7000000
CPUSchedulingPolicy=other
UMask=0007

设置自动启动

sudo systemctl enable turnserver.service

centOS 安装 gitea

发表于 2019-11-08

1. 下载turnserver包

#sudo wget -O gitea https://dl.gitea.io/gitea/1.11.3/gitea-1.11.3-linux-amd64
sudo wget -O gitea https://dl.gitea.io/gitea/master/gitea-master-linux-amd64
sudo chmod +x gitea
cp gitea /usr/local/bin/gitea

sudo adduser --system --shell /bin/bash --comment 'Git Version Control' --user-group --home-dir /home/git -m git
sudo mkdir -p /usr/local/gitea/{custom,data,indexers,public,log}
sudo chown git:git /usr/local/gitea/{data,indexers,log}
sudo chmod 750 /usr/local/gitea/{data,indexers,log}
sudo mkdir /home/gitea
sudo chown root:git /home/gitea
sudo chmod 770 /home/gitea

vim /etc/systemd/system/gitea.service

Using a text editor of your choice, open this newly create file and populate it with the following.

[Unit]
Description=Gitea (Git with a cup of tea)
After=network.target
After=mariadb.service

[Service]
# Modify these two values and uncomment them if you have
# repos with lots of files and get an HTTP error 500 because
# of that
###
#LimitMEMLOCK=infinity
#LimitNOFILE=65535
RestartSec=2s
Type=simple
User=git
Group=git
WorkingDirectory=/var/lib/gitea/
ExecStart=/usr/local/bin/gitea web -c /etc/gitea/app.ini
Restart=always
Environment=USER=git HOME=/home/git GITEA_WORK_DIR=/var/lib/gitea
# If you want to bind Gitea to a port below 1024 uncomment
# the two values below
###
#CapabilityBoundingSet=CAP_NET_BIND_SERVICE
#AmbientCapabilities=CAP_NET_BIND_SERVICE

[Install]
WantedBy=multi-user.target

Enable and start Gitea at boot.

sudo systemctl daemon-reload
sudo systemctl enable gitea
sudo systemctl start gitea

Ensure Gitea is running.

sudo systemctl status gitea

Web-based Gitea configuration.
Finally, open a web browser and point it to:

http://YOUR_SERVER_IP:3000/install
