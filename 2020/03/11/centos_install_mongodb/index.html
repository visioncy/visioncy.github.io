centOS 安装 mongodb

发表于 2020-03-11

1. 添加安装源

vim /etc/yum.repos.d/mongodb-org-4.2.repo

[mongodb-org-4.2]
name=MongoDB Repository
baseurl=https://repo.mongodb.org/yum/redhat/$releasever/mongodb-org/4.2/x86_64/
gpgcheck=1
enabled=1
gpgkey=https://www.mongodb.org/static/pgp/server-4.2.asc

2.安装mongodb常用包

sudo yum install -y mongodb-org

移动数据库

#复制数据库
rsync -av /var/lib/mongo /mnt/database/mongodb/
#备份旧数据库
sudo mv /var/lib/mongo /var/lib/mongo.bak

如果运行数据库是提示没权限，可以进行授权

#Change owner and group
sudo chown -R mongod:mongod /home/mongodb/mongo
#Change context
sudo chcon -R -u system_u -t mongod_var_lib_t /home/mongodb/mongo/

3.修改数据库保存路径
/etc/mongod.conf

初步化数据库
mongo –dbpath /path/

添加超级管理员

mongo

use admin
db.createUser(
  {
    user: "xhadmin",
    pwd: "abc123",
    roles: [ { role: "userAdminAnyDatabase", db: "admin" }, "readWriteAnyDatabase" ]
  }
)


use admin
db.createUser(
  {
    user: "xhadmin",
    pwd: "abc123",
    roles: [ { role: "userAdminAnyDatabase", db: "admin" }, "readWriteAnyDatabase" ]
  }
)

创建数据库用户

use test
db.createUser(
  {
    user: "myTester",
    pwd: "xyz123",
    roles: [ { role: "readWrite", db: "test" },
             { role: "read", db: "reporting" } ]
  }
)

安装步骤参考资料：

https://docs.mongodb.com/manual/tutorial/install-mongodb-on-red-hat/
