centOS 安装 turnserver

1. 下载turnserver包

wget http://www.haproxy.org/download/2.2/src/haproxy-2.2.2.tar.gz
sudo tar -zxvf haproxy-2.2.2.tar.gz
cd haproxy-2.2.2

2. 配置与编译

sudo yum install -y make gcc cc gcc-c++ wget
sudo yum install -y openssl-devel libevent libevent-devel mysql-devel mysql-server
sudo yum install -y libsystemd-dev systemd-devel
make TARGET=linux-glibc PREFIX=/usr/local/haproxy USE_OPENSSL=1 USE_SYSTEMD=1
make install PREFIX=/usr/local/haproxy

cp /usr/local/haproxy/sbin/haproxy /usr/local/sbin/

3. 修改配置文件

vim /etc/haproxy/haproxy.cfg

#---------------------------------------------------------------------
# Defaults settings
#---------------------------------------------------------------------
defaults
    log      global
    option   dontlognull
    option   httpclose
    option   redispatch
    balance  roundrobin
    #maxconn  4096
    timeout  connect 5s
    timeout  client 5s
    timeout  server 3s
    timeout  check 5s
    retries  3
#---------------------------------------------------------------------
# Statistics
#---------------------------------------------------------------------
listen admin_stats
    bind     0.0.0.0:8100
    stats    enable
    mode     http
    log      global
    stats    uri  /stats
    stats    realm Haproxy\ Statistics
    stats    auth  admin:c0s43pa
    stats    admin if TRUE
    stats    refresh 5s
##########################################
frontend stats
    mode     http
    bind *:8404
    http-request use-service prometheus-exporter if { path /metrics }
    stats enable
    stats uri /stats
    stats refresh 10s
#---------------------------------------------------------------------
# IM CLUSTER
#---------------------------------------------------------------------
listen IM_cluster
    bind     0.0.0.0:5222
    mode     tcp
    option   tcplog
    balance  roundrobin
    timeout  client 3h
#    timeout  connect 3s
    timeout  server 3h
    server   im01 47.242.8.252:5222 check inter 5000 rise 2 fall 3

4.设置开机启动

sudo vim /usr/lib/systemd/system/haproxy.service 

[Unit]
Description=HAProxy Load Balancer
After=syslog.target network.target

[Service]
ExecStartPre=/usr/local/sbin/haproxy -f /etc/haproxy/haproxy.cfg   -c -q
ExecStart=/usr/local/sbin/haproxy -Ws -f /etc/haproxy/haproxy.cfg  -p /usr/local/haproxy/haproxy.pid
ExecReload=/bin/kill -USR2 $MAINPID

[Install]
WantedBy=multi-user.target

2.设置开机启动

sudo vim /usr/lib/systemd/system/tomcat.service 

[Unit]
Description=Tomcat-8.5.59
After=syslog.target network.target remote-fs.target nss-lookup.target

[Service]
Type=forking
User=root
Group=root

Restart=always

WantedBy=multi-user.target

centOS 安装 turnserver

1. 下载turnserver包

wget http://turnserver.open-sys.org/downloads/v4.5.1.1/turnserver-4.5.1.1.tar.gz 
sudo tar -zxvf turnserver-4.5.1.1.tar.gz
cd turnserver-4.5.1.1 && sudo ./configure

2. 配置与编译

sudo yum install -y make gcc cc gcc-c++ wget
sudo yum install -y openssl-devel libevent libevent-devel mysql-devel mysql-server
sudo make && sudo make install

3. 修改配置文件

sudo cp /usr/local/etc/turnserver.conf.default   /usr/local/etc/turnserver.conf
sudo vim /usr/local/etc/turnserver.conf

在文件顶部加入如下配置 

listening-port=3478
user=mangoturn:cvKuUdaQ
realm=turn02.imangoim.com

附：手去启动TURN命令

turnserver -c /usr/local/etc/turnserver.conf -o –v

4.设置开机启动

sudo vim /etc/systemd/system/turnserver.service

文件内容

[Unit]
Description=coturn
Documentation=man:coturn(1) man:turnadmin(1) man:turnserver(1)
After=syslog.target network.target

[Service]
#User=turnserver
#Group=turnserver
Type=forking
#EnvironmentFile=/etc/sysconfig/turnserver
PIDFile=/var/run/turnserver.pid
ExecStart=/usr/local/bin/turnserver -o -c /usr/local/etc/turnserver.conf $EXTRA_OPTIONS
ExecStopPost=/usr/bin/rm -f /var/run/turnserver.pid
Restart=on-abort

LimitCORE=infinity
LimitNOFILE=999999
LimitNPROC=60000
LimitRTPRIO=infinity
LimitRTTIME=7000000
CPUSchedulingPolicy=other
UMask=0007

设置自动启动

sudo systemctl enable turnserver.service
