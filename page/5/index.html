centOS 安装 turnserver

wget https://mirrors.tuna.tsinghua.edu.cn/apache/tomcat/tomcat-8/v8.5.59/bin/apache-tomcat-8.5.59.tar.gz
sudo tar -zxvf apache-tomcat-8.5.59.tar.gz
mv apache-tomcat-8.5.59 /usr/local/tomcat

2.设置开机启动

sudo vim /usr/lib/systemd/system/tomcat.service 

[Unit]
Description=Tomcat-8.5.59
After=syslog.target network.target remote-fs.target nss-lookup.target

[Service]
Type=forking
User=root
Group=root
Environment="JAVA_HOME=/usr/local/jdk/"
ExecStart=/usr/local/tomcat/bin/startup.sh
ExecReload=/usr/local/tomcat/bin/startup.sh
ExecStop=/usr/local/tomcat/bin/shutdown.sh
SuccessExitStatus=143
Restart=always
RestartSec=5
StartLimitInterval=0
RestartPreventExitStatus=143 137 SIGTERM SIGKILL

[Install]
WantedBy=multi-user.target

centOS 安装 turnserver

1. 下载turnserver包

wget http://turnserver.open-sys.org/downloads/v4.5.1.1/turnserver-4.5.1.1.tar.gz 
sudo tar -zxvf turnserver-4.5.1.1.tar.gz
cd turnserver-4.5.1.1 && sudo ./configure

2. 配置与编译

sudo yum install -y make gcc cc gcc-c++ wget
sudo yum install -y openssl-devel libevent libevent-devel mysql-devel mysql-server
sudo make && sudo make install

3. 修改配置文件

sudo cp /usr/local/etc/turnserver.conf.default   /usr/local/etc/turnserver.conf
sudo vim /usr/local/etc/turnserver.conf

在文件顶部加入如下配置 

listening-port=3478
user=mangoturn:cvKuUdaQ
realm=turn02.imangoim.com

附：手去启动TURN命令

turnserver -c /usr/local/etc/turnserver.conf -o –v

4.设置开机启动

sudo vim /etc/systemd/system/turnserver.service

文件内容

[Unit]
Description=coturn
Documentation=man:coturn(1) man:turnadmin(1) man:turnserver(1)
After=syslog.target network.target

[Service]
#User=turnserver
#Group=turnserver
Type=forking
#EnvironmentFile=/etc/sysconfig/turnserver
PIDFile=/var/run/turnserver.pid
ExecStart=/usr/local/bin/turnserver -o -c /usr/local/etc/turnserver.conf $EXTRA_OPTIONS
ExecStopPost=/usr/bin/rm -f /var/run/turnserver.pid
Restart=on-abort

LimitCORE=infinity
LimitNOFILE=999999
LimitNPROC=60000
LimitRTPRIO=infinity
LimitRTTIME=7000000
CPUSchedulingPolicy=other
UMask=0007

设置自动启动

sudo systemctl enable turnserver.service

kms 激活 windows office

Windows

Office

错误：The Software Licensing Service reported that the product SKU is not found

这不是vlmcsd中的错误，这是Office的一个问题。可能是您的系统上没有安装批量许可SKU，打开C:\Program Files\Microsoft Office\root\icenses16文件夹，在这个文件夹中应该是一堆XRM-MS文件。如果是这样的，下面的批处理文件会帮你安装许可证证书：

转换VOL版本批处理

@echo off
:ADMIN
openfiles >nul 2>nul ||(
echo CreateObject^("Shell.Application"^).ShellExecute "%~s0", "", "", "runas", 1 >> "%temp%\getadmin.vbs"
"%temp%\getadmin.vbs" >nul 2>&1
goto:eof
)
del /f /q "%temp%\getadmin.vbs" >nul 2>nul

for /f "tokens=6 delims=[]. " %%G in ('ver') do set win=%%G

setlocal

set OPPKEY=XQNVK-8JYDB-WJ9W3-YJ8YR-WFG99
set PPKEY=YG9NW-3K39V-2T3HJ-93F3Q-G83KT
set VPKEY=PD3PC-RHNGV-FXJ29-8JK7D-RJRJK
set S4BKEY=869NQ-FJ69K-466HW-QYCP2-DDBV6

pushd "%~dp0"
Title Office 2016 Retail to Volume License Converter

SET OfficePath=%ProgramFiles%\Microsoft Office
if not exist "%OfficePath%\root\Licenses16" SET OfficePath=%ProgramFiles(x86)%\Microsoft Office
if not exist "%OfficePath%\root\Licenses16" (
    echo Could not find the license files for Office 2016!
    pause
    goto :eof
)

echo Press Enter to start VL-Conversion...
echo.
pause
echo.
cd /D "%SystemRoot%\System32"

if %win% GEQ 9200 (
    cscript slmgr.vbs /ilc "%OfficePath%\root\Licenses16\ProPlusVL_KMS_Client-ppd.xrm-ms"
    cscript slmgr.vbs /ilc "%OfficePath%\root\Licenses16\ProPlusVL_KMS_Client-ul.xrm-ms"
    cscript slmgr.vbs /ilc "%OfficePath%\root\Licenses16\ProPlusVL_KMS_Client-ul-oob.xrm-ms"

    cscript slmgr.vbs /ilc "%OfficePath%\root\Licenses16\ProjectProVL_KMS_Client-ul-oob.xrm-ms"

git 代理配置方式 

# 设置ss
git config --global http.proxy 'socks5://127.0.0.1:1080'
git config --global https.proxy 'socks5://127.0.0.1:1080'

# 设置代理
git config --global https.proxy http://127.0.0.1:1080
git config --global https.proxy https://127.0.0.1:1080

# 取消代理
git config --global --unset http.proxy
git config --global --unset https.proxy

补充说明：socks5和socks5h的区别

socks5：适合本地能够解析目标主机域名(比如github.com)但是访问速度慢,来提高下载速度

socks5h：用与本地不能解析目标主机域名(比如google),由代理服务器解析目标主机域名
